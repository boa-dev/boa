Module icu_plurals::rules::reference::ast

source · [−]
Expand description

AST provides a set of Syntax Tree Nodes used to store the output of parse method that is used in test_condition method to evaluate whether a given PluralCategory should be used.

Examples

use icu::plurals::rules::reference::parse_condition;
use icu::plurals::rules::reference::ast::*;

let input = "i = 1";

let ast = parse_condition(input.as_bytes())
    .expect("Parsing failed.");

assert_eq!(ast, Condition(vec![
    AndCondition(vec![
        Relation {
            expression: Expression {
                operand: Operand::I,
                modulus: None,
            },
            operator: Operator::Eq,
            range_list: RangeList(vec![
                RangeListItem::Value(
                    Value(1)
                )
            ])
        }
    ])
]));
